{{- /* Hugo British Scout Group theme CC BY-SA 4.0 © Peter Mahon https://github.com/petemahon/hugo-british-scout-group gallery-photo-og — Open Graph meta block for per-photo pages. Called from partials/head.html when .Layout == "photo". Emits og:title (gallery title + optional caption), og:description (the photo's alt text), og:url, og:type and og:image (the 1280w WebP resize). When this partial runs, head.html suppresses its default og:title / og:description / og:url / og:image emission so we never produce duplicate tags. */ -}} {{- $gallery := .CurrentSection -}} {{- $slug := .Params.gallery_slug -}} {{- $file := .Params.image_file -}} {{- $imgPath := printf "galleries/%s/%s" $slug $file -}} {{- $img := resources.Get $imgPath -}} {{- $caption := .Params.image_caption -}} {{- $alt := .Params.image_alt -}} {{- $title := $gallery.Title -}} {{- if ne $caption "" -}}{{- $title = printf "%s — %s" $gallery.Title $caption -}}{{- end -}} {{- $desc := $alt -}} {{- if eq $desc "" -}}{{- $desc = printf "Photo from %s" $gallery.Title -}}{{- end -}} {{- with $img }} {{- $og := .Resize "1280x webp q82" }} {{- end }}
